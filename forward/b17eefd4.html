жҢҮй’Ҳе’Ңең°еқҖзҡ„еҢәеҲ«пјҡ

  • ең°еқҖпјҡе°ұжҳҜеҶ…еӯҳең°еқҖ(з”Ёеӯ—иҠӮжқҘжҸҸиҝ°зҡ„еҶ…еӯҳең°еқҖ)

	return &Student{

	}

func NewCollege() *College {
	return &College{
		AllStudent: make([]*Student,0,100),
 //еҲқе§ӢеҢ–з»“жһ„дҪ“дёӯзҡ„еҲҮзүҮ
	}

//ж·»еҠ еӯҰз”ҹ
func (c *College) add(s *Student) {
	for _,v := range c.AllStudent {
		if v.Name == s.Name {
			fmt.Printf("%vе·ІеӯҳеңЁ",s.Name)
			return
		}
	}
	c.AllStudent = append(c.AllStudent,s)
}

//еҲ йҷӨеӯҰз”ҹ
func (c *College) del(s *Student) {
	for index ,v := range c.AllStudent {
		if v.Name == s.Name {
			c.AllStudent = append(c.AllStudent[:index],c.AllStudent[index+1:]...)
		}
	}
	fmt.Printf("%vдёҚеӯҳеңЁ",s.Name)
}

	for index,v := range c.AllStudent {
		if v.Name == s.Name {

		}
	}
	fmt.Printf("%vдёҚеӯҳеңЁ",s.Name)
}
